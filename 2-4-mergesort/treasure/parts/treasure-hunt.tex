\begin{frame}{}
  \fignocaption{width = 0.35\textwidth}{figs/tip-4}
  {\vspace{-0.30cm} \centerline{\textcolor{cyan}{\small (MON, April 16, 2018)}}}

  \vspace{0.50cm}
  \teal{$\textsc{Merge}(A, B)$:} keeps comparing the smallest remaining elements in $A$ and $B$
  {\small \textcolor{gray}{(and removing the minimum)}}, \red{\it until} one of them is empty.

  \begin{definition}[$R$]
    \red{$R$} is the number of elements left in the non-empty array by \teal{$\textsc{Merge}(A,B)$}
    on $A$ of size $a$ and $B$ of size $b$.
  \end{definition}

  \begin{theorem}
    \vspace{-0.50cm}
    \[
      Pr(R \ge r) = \frac{\binom{a + b - r}{a}}{\binom{a + b}{a}} + \frac{\binom{a + b - r}{b}}{\binom{a + b}{a}},
      \; \mathbb{E}(R) = \sum_{r} Pr(R \ge r) = \frac{a}{b+1} + \frac{b}{a+1}
    \]
  \end{theorem}
\end{frame}
